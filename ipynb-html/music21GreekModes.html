def returnScale(mytonic='c', mymode='major', ownKey = False):
    
    modes = ['ionian','dorian','phrygian','lydian','mixolidian','aeolian','locrian','major','minor']
    
    if mymode in modes:
        
        if mymode == 'ionian':
            mymode = 'major'
        elif mymode == 'aeolian':
            mymode = 'minor'
            
        myfunc = eval("scale.{0}Scale".format(mymode.capitalize()))

    else:
        mymode = 'major'
        myfunc = scale.MajorScale
        
    pitches = myfunc(mytonic).getPitches()

    mystream2 = stream.Stream()
    
    if ownKey:
        mystream2.append(key.Key(mytonic,mymode))

    for x in range(len(mystream2), len(pitches)):
        mynote = note.Note(pitches[x])

        mystream2.append(mynote)

        mystream2[x].addLyric(interval.notesToInterval(pitches[0], mynote).directedName)
        mystream2[x].addLyric(interval.notesToInterval(pitches[(x-1 if x>0 else 0)], mynote).directedName)
        
    return mystream2
